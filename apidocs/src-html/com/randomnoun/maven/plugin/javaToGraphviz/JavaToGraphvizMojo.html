167    // private method in FileSetManager
168
169        private DirectoryScanner scan(FileSet fileSet) {
170                File basedir = new File(fileSet.getDirectory());
171                if (!basedir.exists() || !basedir.isDirectory()) {
172                        return null;
173                }
174
175                DirectoryScanner scanner = new DirectoryScanner();
176
177                String[] includesArray = fileSet.getIncludesArray();
178                String[] excludesArray = fileSet.getExcludesArray();
179
180                if (includesArray.length > 0) {
181                        scanner.setIncludes(includesArray);
182                }
183
184                if (excludesArray.length > 0) {
185                        scanner.setExcludes(excludesArray);
186                }
187
188                if (fileSet.isUseDefaultExcludes()) {
189                        scanner.addDefaultExcludes();
190                }
191
192                scanner.setBasedir(basedir);
193                scanner.setFollowSymlinks(fileSet.isFollowSymlinks());
194
195                scanner.scan();
196
197                return scanner;
198        }

203    /**
204     * Generates the JAR.
205     * 
206     * @throws MojoExecutionException in case of an error.
207     */
208        public void createGraphviz() throws MojoExecutionException {
209                if (outputDirectory == null) {
210                        throw new IllegalArgumentException("Missing outputDirectory");
211                }
212                if (outputFilenamePattern == null) {
213                        throw new IllegalArgumentException("Missing outputFilename or outputFilenamePattern");
214                }
215                getLog().info("Creating graphviz diagrams");
216
217
218                DirectoryScanner scanner = scan(fileset);
219                String[] files = scanner.getIncludedFiles(); // also performs exclusion. So way to go, maven.
220
221                try {
222                    
223                    for (String filename : files) {
224                        
225                        File dir = new File(filename).getParentFile();
226                        //getLog().info("filename " + filename + " path " + dir.getPath());
227                        //getLog().info("i " + invokeGraphviz + " path " + graphvizExecutable.getCanonicalPath());
228                        
229                        File f = new File(fileset.getDirectory(), filename);
230                        
231                JavaToGraphviz javaToGraphviz = new JavaToGraphviz();
232                javaToGraphviz.setFormat("dot");
233                javaToGraphviz.setSourceVersion(source);
234                if (baseCssUrl != null) {
235                    javaToGraphviz.setBaseCssUrl(baseCssUrl);
236                }
237                javaToGraphviz.setUserCssUrls(userCssUrls);
238                javaToGraphviz.setUserCssRules(userCssRules);
239                javaToGraphviz.setOptions(options);
240              
241                getLog().info("Reading " + f.getCanonicalPath());            
242                FileInputStream is = new FileInputStream(f);
243                javaToGraphviz.parse(is, "UTF-8");
244                is.close();
245                String n = f.getName();
246                if (n.indexOf(".") != -1) { n = n.substring(0, n.indexOf(".")); }
247            
248                int diagramIndex = 0;
249                boolean hasNext;
250                do {
251                    String on = outputFilenamePattern;
252
253                    on = Text.replaceString(on, "{directory}", dir.getPath());
254                    on = Text.replaceString(on, "{basename}", n);
255                    on = Text.replaceString(on, "{index}", String.valueOf(diagramIndex));
256                    getLog().info("Writing " + on); 
257                    File destFile = new File(outputDirectory, on);
258                    
259                    if (destFile.exists() && destFile.isDirectory()) {
260                        throw new MojoExecutionException(destFile + " is a directory.");
261                    }
262                    if (destFile.exists() && !destFile.canWrite()) {
263                        throw new MojoExecutionException(destFile + " is not writable.");
264                    }
265                    if (!destFile.getParentFile().exists()) {
266                        // create the parent directory...
267                        if (!destFile.getParentFile().mkdirs()) {
268                            // Failure, unable to create specified directory for some unknown reason.
269                            throw new MojoExecutionException("Unable to create directory or parent directory of " + destFile);
270                        }
271                    }
272                    if (destFile.exists()) {
273                        // delete existing
274                        if (!destFile.delete()) {
275                            throw new MojoExecutionException("Unable to delete existing file " + destFile);
276                        }
277                    }
278                    
279                    
280                    FileOutputStream fos = new FileOutputStream(destFile);
281                    PrintWriter pw = new PrintWriter(fos);
282                    hasNext = javaToGraphviz.writeGraphviz(pw);
283                    pw.flush();
284                    fos.close();
285                    
286                    if (invokeGraphviz) {
287                        getLog().info("invoking graphviz");
288                        if (on.indexOf(".")==-1) {
289                            on = on + ".png";
290                        } else {
291                            on = on.substring(0, on.lastIndexOf(".")) + ".png";
292                        }
293                        File destPngFile = new File(outputDirectory, on);
294                        
295                        String output;
296                        try {
297                            output = ProcessUtil.exec(new String[] {
298                                graphvizExecutable.getCanonicalPath(),
299                                destFile.getCanonicalPath(),
300                                "-Tpng",
301                                "-o" + destPngFile.getCanonicalPath()
302                            });
303                        } catch (ProcessException e) {
304                            getLog().info("graphviz process failed", e);
305                        }
306                    }
307                        
308                    diagramIndex++;
309                } while (hasNext);
310                    }
311                
312                } catch (IOException ioe) {
313                        // trouble at the mill
314                        throw new MojoExecutionException("Could not create graphviz diagram", ioe); 
315                }
316
317        }
